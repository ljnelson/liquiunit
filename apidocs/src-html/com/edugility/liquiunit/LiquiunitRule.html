239       * <!-- Formatting here is important for proper Javadoc output -->
240       * <blockquote><pre>new CompositeResourceAccessor(new URLResourceAccessor(Thread.currentThread().getContextClassLoader()),

242       *

244       * DataSource#getConnection(String, String) acquire a

248       * @param username the username to use during {@linkplain

252       * @param password the password to use during {@linkplain
253       * DataSource#getConnection(String, String) <code>Connection</code>

256       * @param contexts a variable number of Liquibase <a
257       * href="http://www.liquibase.org/documentation/contexts.html">contexts</a>;

261       *

263       *
264       * @see #setPassword(String)
265       *
266       * @see #setContexts(Iterable)
267       *
268       * @see #setResourceAccessor(ResourceAccessor)
269       *
270       * @see #setChangeLogResourceName(String)
271       */
272      public LiquiunitRule(final DataSource dataSource, final String username, final String password, final String... contexts) {
273        super();
274        this.logger = LogFactory.getInstance().getLog("liquiunit");
275        assert this.logger != null;
276        this.dataSource = dataSource;
277        this.setUsername(username);
278        this.setPassword(password);
279        this.setChangeLogResourceName("changelog.xml");
280        this.setResourceAccessor(new CompositeResourceAccessor(new URLResourceAccessor(Thread.currentThread().getContextClassLoader()), new FileSystemResourceAccessor(System.getProperty("user.dir"))));
281        if (contexts != null && contexts.length > 0) {
282          this.setContexts(Arrays.asList(contexts));
283        }
284      }

448      /**
449       * Transforms the return value of the {@link #getContexts()} into a
450       * comma-separated {@link String} suitable for passing to the {@link
451       * Liquibase#update(String)} method and returns it.
452       *
453       * @return a comma-separated {@link String} of Liquibase <a
454       * href="http://www.liquibase.org/documentation/contexts.html">contexts</a>,
455       * or {@code null}
456       *
457       * @see #getContexts()
458       *
459       * @see #setContexts(Iterable)
460       *
461       * @see Liquibase#update(String)
462       */
463      private final String getContextsString() {
464        final StringBuilder sb;
465        final Iterable<? extends String> contexts = this.getContexts();
466        if (contexts == null) {
467          sb = null;
468        } else {        
469          final Iterator<? extends String> iterator = contexts.iterator();
470          if (iterator == null) {
471            sb = null;
472          } else {
473            sb = new StringBuilder();
474            while (iterator.hasNext()) {
475              final String context = iterator.next();
476              if (context != null) {
477                sb.append(context);
478                if (iterator.hasNext()) {
479                  sb.append(",");
480                }
481              }
482            }
483          }
484        }
485    
486        final String returnValue;
487        if (sb == null) {
488          returnValue = null;
489        } else {
490          returnValue = sb.toString();
491        }
492        return returnValue;
493      }

495      /**
496       * Overrides the {@link ExternalResource#before()} method to
497       * {@linkplain DataSource#getConnection(String, String) acquire a
498       * <code>Connection</code>}, {@linkplain
499       * DatabaseFactory#findCorrectDatabaseImplementation(DatabaseConnection)
500       * find the proper Liquibase <code>Database</code> implementation},
501       * {@linkplain #createLiquibase(Database) create a
502       * <code>Liquibase</code> instance} and, using it, {@linkplain
503       * Liquibase#update(String) update} the backing database using the
504       * {@linkplain #getChangeLogResourceName() affiliated changelog}.
505       *
506       * @exception LiquibaseException if there was a Liquibase-related
507       * error
508       *
509       * @exception SQLException if there was a database-related error
510       * 
511       * @see DataSource#getConnection(String, String)
512       *
513       * @see #createLiquibase(Database)
514       *
515       * @see #shouldUpdate(Liquibase)
516       *
517       * @see #getContexts()
518       *
519       * @see Liquibase#update(String)
520       */
521      @Override
522      protected void before() throws LiquibaseException, SQLException {
523        if (this.dataSource != null) {
524    
525          this.c = this.dataSource.getConnection(this.getUsername(), this.getPassword());
526          if (this.c == null) {
527            throw new IllegalStateException("this.dataSource.getConnection()", new NullPointerException("this.dataSource.getConnection()"));
528          } else if (!this.c.isValid(0)) {
529            throw new IllegalStateException("!this.c.isValid()");
530          }
531    
532          final DatabaseFactory databaseFactory = DatabaseFactory.getInstance();
533          assert databaseFactory != null;
534          final Database database = databaseFactory.findCorrectDatabaseImplementation(new JdbcConnection(this.c));
535    
536          this.liquibase = this.createLiquibase(database);
537          if (this.liquibase != null && this.shouldUpdate(liquibase)) {
538            try {
539              this.liquibase.update(this.getContextsString());
540            } finally {
541              this.liquibase.forceReleaseLocks();
542            }
543          }
544        }
545      }

547      /**
548       * Tries to {@linkplain ClassLoader#getResource(String) load} the
549       * {@linkplain #getChangeLogResourceName() classpath resource
550       * identifying a Liquibase changelog}, and, if successful, creates a
551       * new {@link Liquibase} instance initialized to read from it, and
552       * returns it.
553       *
554       * <p>This method may return {@code null}.</p>
555       *
556       * @param database the {@link Database} to operate on; must not be
557       * {@code null}
558       *
559       * @return a new configured {@link Liquibase} instance, or {@code
560       * null} if there is no changelog to read and hence no reason to
561       * construct a {@link Liquibase} instance
562       *
563       * @exception LiquibaseException if a problem occurred while
564       * constructing the new {@link Liquibase} instance
565       *
566       * @see #changeLogResourceExists()
567       *
568       * @see #getResourceAccessor()
569       */
570      protected Liquibase createLiquibase(final Database database) throws LiquibaseException {
571        final Liquibase liquibase;
572        if (changeLogResourceExists()) {
573          liquibase = null;
574        } else {
575          liquibase = new Liquibase(changeLogResourceName, this.getResourceAccessor(), database);
576        }
577        return liquibase;
578      }

580      /**
581       * Returns {@code true} if the <a
582       * href="http://www.liquibase.org/documentation/databasechangelog.html">changelog</a>
583       * identified by the return value of the {@link
584       * #getChangeLogResourceName()} method actually exists.
585       *
586       * @return {@code true} if the <a
587       * href="http://www.liquibase.org/documentation/databasechangelog.html">changelog</a>
588       * identified by the return value of the {@link
589       * #getChangeLogResourceName()} method actually exists; {@code
590       * false} otherwise
591       *
592       * @see #getChangeLogResourceName()
593       */
594      protected boolean changeLogResourceExists() {
595        final boolean returnValue;
596        final String changeLogResourceName = this.getChangeLogResourceName();
597        if (changeLogResourceName == null) {
598          returnValue = false;
599        } else {
600          ClassLoader cl = Thread.currentThread().getContextClassLoader();
601          if (cl == null) {
602            cl = ClassLoader.getSystemClassLoader();
603            if (cl == null) {
604              cl = this.getClass().getClassLoader();
605            }
606          }
607          returnValue = cl != null && cl.getResource(changeLogResourceName) != null;
608          this.logger.debug("ChangeLog \"" + changeLogResourceName + "\" exists? " + returnValue);
609        }
610        return returnValue;
611      }

669       * {@code null} in which case {@code false} will be returned

673       * called
674       *
675       * @exception LiquibaseException if there was a Liquibase-related
676       * error
677       *
678       * @exception SQLException if there was a database-related error
679       *
680       * @see Liquibase#update(String)
681       */
682      protected boolean shouldUpdate(final Liquibase liquibase) throws LiquibaseException, SQLException {
683        boolean returnValue = liquibase != null && !"false".equals(System.getProperty(Liquibase.SHOULD_RUN_SYSTEM_PROPERTY));
684        if (returnValue) {
685          final Database database = liquibase.getDatabase();
686          if (database != null) {
687            final ChangeLogHistoryService changeLogHistoryService = ChangeLogHistoryServiceFactory.getInstance().getChangeLogService(database);
688            returnValue = !(changeLogHistoryService instanceof StandardChangeLogHistoryService) || !((StandardChangeLogHistoryService)changeLogHistoryService).hasDatabaseChangeLogTable();
689          }
690        }
691        return returnValue;
692      }

694      /**
695       * Overrides the {@link ExternalResource#after()} method to
696       * {@linkplain Connection#close() close} the {@link Connection} that
697       * was {@linkplain DataSource#getConnection(String, String)
698       * acquired} during the execution of the {@link #before()} method.
699       *
700       * @see #before()
701       *
702       * @see Connection#close()
703       */
704      @Override
705      protected void after() {
706        if (this.c != null) {
707          try {
708            this.c.close();
709          } catch (final SQLException ignore) {
710            // ignore
711          }
712          this.c = null;
713        }
714      }
