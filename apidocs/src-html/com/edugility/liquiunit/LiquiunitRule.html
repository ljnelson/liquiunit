188  /**
189   * Creates a new {@link LiquiunitRule}.
190   *
191   * <p>The new {@link LiquiunitRule} will have {@linkplain
192   * #getChangeLogResourceNames() changelog resource names} of "{@code
193   * META-INF/liquibase/changelog.xml}" and "{@code
194   * META-INF/liquibase/test-changelog.xml}", and a {@link
195   * #getResourceAccessor() ResourceAccessor} {@linkplain
196   * #setResourceAccessor(ResourceAccessor) initialized} with the
197   * following code:</p>
198   *
199   * <!-- Formatting here is important for proper Javadoc output -->
200   * <blockquote><pre>final ResourceAccessor classLoaderAccessor = new ClassLoaderResourceAccessor(Thread.currentThread().getContextClassLoader());
201   *final ResourceAccessor urlAccessor = new URLResourceAccessor(classLoaderAccessor);
202   *final ResourceAccessor fileAccessor = new FileSystemResourceAccessor(System.getProperty("user.dir"));
203   *new CompositeResourceAccessor(urlAccessor, classLoaderAccessor, fileAccessor);</pre></blockquote>
204   *
205   * <p>Consider passing an {@link H2Rule} as the value for the {@code
206   * dataSource} parameter for thread-safe parallel in-memory database
207   * testing scenarios.</p>
208   *
209   * @param dataSource the {@link DataSource} to use to {@linkplain
210   * DataSource#getConnection(String, String) acquire a
211   * <code>Connection</code>}; may be {@code null} in which case this
212   * {@link LiquiunitRule} will effectively do nothing
213   *
214   * @param contexts a variable number of Liquibase <a
215   * href="http://www.liquibase.org/documentation/contexts.html">contexts</a>;
216   * may be {@code null}
217   *
218   * @see DataSource#getConnection(String, String)
219   *
220   * @see #setContexts(Iterable)
221   *
222   * @see #setResourceAccessor(ResourceAccessor)
223   *
224   * @see #setChangeLogResourceNames(Iterable)
225   */
226  public LiquiunitRule(final DataSource dataSource, final String... contexts) {
227    super();
228    this.logger = LogFactory.getInstance().getLog("liquiunit");
229    assert this.logger != null;
230    String logLevel = System.getProperty("liquiunit.logLevel", "").trim();
231    if (!logLevel.isEmpty()) {
232      this.logger.setLogLevel(logLevel);
233    }
234    logLevel = System.getProperty("liquibase.logLevel", "").trim();
235    if (!logLevel.isEmpty()) {
236      final Logger logger = LogFactory.getInstance().getLog();
237      if (logger != null) {
238        logger.setLogLevel(logLevel);
239      }
240    }
241    this.logger.debug("Entering LiquiunitRule(DataSource, String[]); parameters: dataSource = " + dataSource + "; contexts = " + (contexts == null ? "null" : Arrays.asList(contexts)));
242    this.dataSource = dataSource;
243    final List<String> changeLogResourceNames = new ArrayList<String>();
244    changeLogResourceNames.add("META-INF/liquibase/changelog.xml");
245    changeLogResourceNames.add("META-INF/liquibase/test-changelog.xml");
246    this.setChangeLogResourceNames(changeLogResourceNames);
247    final ResourceAccessor classLoaderAccessor = new ClassLoaderResourceAccessor(Thread.currentThread().getContextClassLoader());
248    final ResourceAccessor urlAccessor = new URLResourceAccessor(classLoaderAccessor);
249    final ResourceAccessor fileAccessor = new FileSystemResourceAccessor(System.getProperty("user.dir"));
250    this.setResourceAccessor(new CompositeResourceAccessor(urlAccessor, classLoaderAccessor, fileAccessor));
251    if (contexts != null && contexts.length > 0) {
252      this.setContexts(Arrays.asList(contexts));
253    }
254    this.logger.debug("Exiting LiquiunitRule(DataSource, String[])");
255  }

362   * Transforms the return value of the {@link #getContexts()} into a

376  private final String getContextsString() {
377    this.logger.debug("Entering getContextsString()");
378    final StringBuilder sb;
379    final Iterable<? extends String> contexts = this.getContexts();
380    if (contexts == null) {
381      sb = null;
382    } else {        
383      final Iterator<? extends String> iterator = contexts.iterator();
384      if (iterator == null) {
385        sb = null;
386      } else {
387        sb = new StringBuilder();
388        while (iterator.hasNext()) {
389          final String context = iterator.next();
390          if (context != null) {
391            sb.append(context);
392            if (iterator.hasNext()) {
393              sb.append(",");
394            }
395          }
396        }
397      }
398    }
399
400    final String returnValue;
401    if (sb == null) {
402      returnValue = null;
403    } else {
404      returnValue = sb.toString();
405    }
406    this.logger.debug("Exiting getContextsString(); returning: " + returnValue);
407    return returnValue;
408  }

410  /**
411   * Overrides the {@link ExternalResource#before()} method to
412   * {@linkplain DataSource#getConnection(String, String) acquire a
413   * <code>Connection</code>}, {@linkplain
414   * DatabaseFactory#findCorrectDatabaseImplementation(DatabaseConnection)
415   * find the proper Liquibase <code>Database</code> implementation},
416   * {@linkplain #createLiquibase(Database) create a
417   * <code>Liquibase</code> instance} and, using it, {@linkplain
418   * Liquibase#update(String) update} the backing database using the
419   * {@linkplain #getChangeLogResourceNames() affiliated changelogs}.
420   *
421   * @exception LiquibaseException if there was a Liquibase-related
422   * error
423   *
424   * @exception SQLException if there was a database-related error
425   * 
426   * @see DataSource#getConnection(String, String)
427   *
428   * @see #createLiquibase(Database)
429   *
430   * @see #shouldUpdate(Liquibase)
431   *
432   * @see #getContexts()
433   *
434   * @see Liquibase#update(String)
435   */
436  @Override
437  protected void before() throws LiquibaseException, SQLException {
438    this.logger.debug("Entering before()");
439    if (this.dataSource != null) {
440
441      final Connection c = this.dataSource.getConnection();
442      if (c == null) {
443        throw new IllegalStateException("this.dataSource.getConnection()", new NullPointerException("this.dataSource.getConnection()"));
444      } else if (!c.isValid(0)) {
445        throw new IllegalStateException("!c.isValid()");
446      }
447      final JdbcConnection jc = new JdbcConnection(c);
448      try {
449        final DatabaseFactory databaseFactory = DatabaseFactory.getInstance();
450        assert databaseFactory != null;
451        final Database database = databaseFactory.findCorrectDatabaseImplementation(jc);
452        
453        this.liquibase = this.createLiquibase(database);
454        if (this.liquibase != null && this.shouldUpdate(liquibase)) {
455          try {
456            this.liquibase.update(this.getContextsString());
457          } finally {
458            this.liquibase.forceReleaseLocks();
459          }
460        }
461      } finally {
462        try {
463          jc.close();
464        } catch (final DatabaseException ignore) {
465          
466        }
467      }
468
469    }
470    this.logger.debug("Exiting before()");
471  }

473  /**
474   * Tries to {@linkplain ClassLoader#getResource(String) load} the
475   * {@linkplain #getChangeLogResourceNames() classpath resources
476   * identifying Liquibase changelogs}, and, if successful, creates a
477   * new {@link Liquibase} instance initialized to read from them, and
478   * returns it.
479   *
480   * <p>This method may return {@code null}.</p>
481   *
482   * <h2>Implementation Notes</h2>
483   *
484   * <p>When a {@link Liquibase} instance is asked to {@linkplain
485   * Liquibase#update(Contexts, LabelExpression) perform an update
486   * operation}, it operates on a single {@link DatabaseChangeLog}.
487   * But this class can specify {@linkplain
488   * #getChangeLogResourceNames() several changelogs to read}.
489   * Consequently, the {@link Liquibase} instance constructed by this
490   * method uses an internally built composite {@link
491   * DatabaseChangeLog} that logically aggregates all appropriate
492   * changelogs by virtue of its {@link
493   * DatabaseChangeLog#include(String, boolean, ResourceAccessor)}
494   * method.  From the point of view of a {@link Liquibase} instance
495   * constructed by this method, therefore, there is only one {@link
496   * DatabaseChangeLog} in existence that has been formed by the
497   * inclusion of {@linkplain #getChangeLogResourceNames() all
498   * relevant existing changelogs} into one.</p>
499   *
500   * @param database the {@link Database} to operate on; must not be
501   * {@code null}
502   *
503   * @return a new configured {@link Liquibase} instance, or {@code
504   * null} if there is no changelog to read and hence no reason to
505   * construct a {@link Liquibase} instance
506   *
507   * @exception LiquibaseException if a problem occurred while
508   * constructing the new {@link Liquibase} instance
509   *
510   * @see #changeLogResourceExists()
511   *
512   * @see #getResourceAccessor()
513   *
514   * @see Liquibase
515   *
516   * @see Liquibase#Liquibase(DatabaseChangeLog, ResourceAccessor, Database)
517   *
518   * @see DatabaseChangeLog
519   *
520   * @see DatabaseChangeLog#include(String, boolean, ResourceAccessor)
521   */
522  protected Liquibase createLiquibase(final Database database) throws LiquibaseException {
523    this.logger.debug("Entering createLiquibase(Database); parameters: database = " + database);
524    Liquibase liquibase = null;
525    if (changeLogResourceExists()) {
526      final Iterable<? extends String> changeLogResourceNames = this.getChangeLogResourceNames();
527      if (changeLogResourceNames != null) {
528        final ResourceAccessor resourceAccessor = this.getResourceAccessor();        
529        AugmentableDatabaseChangeLog changeLog = null;
530        for (final String changeLogResourceName : changeLogResourceNames) {
531          if (changeLogResourceName != null && this.changeLogResourceExists(changeLogResourceName)) {
532            if (changeLog == null) {
533              assert liquibase == null;
534              changeLog = this.createAugmentableDatabaseChangeLog(database, changeLogResourceName);
535              liquibase = new Liquibase(changeLog, resourceAccessor, database);
536              changeLog.setChangeLogParameters(liquibase.getChangeLogParameters());
537            } else {
538              assert liquibase != null;
539              changeLog.include(changeLogResourceName, resourceAccessor);
540            }
541          }
542        }
543      }
544    }
545    this.logger.debug("Exiting createLiquibase(Database); returning: " + liquibase);
546    return liquibase;
547  }

549  /**
550   * Returns {@code true} if at least one of the <a
551   * href="http://www.liquibase.org/documentation/databasechangelog.html">changelogs</a>
552   * identified by the return value of the {@link
553   * #getChangeLogResourceNames()} method actually exists.
554   *
555   * @return {@code true} if at least one of the <a
556   * href="http://www.liquibase.org/documentation/databasechangelog.html">changelogs</a>
557   * identified by the return value of the {@link
558   * #getChangeLogResourceNames()} method actually exists; {@code
559   * false} otherwise
560   *
561   * @see #getChangeLogResourceNames()
562   */
563  protected boolean changeLogResourceExists() {
564    this.logger.debug("Entering changeLogResourceExists()");
565    final boolean returnValue;
566    final Iterable<? extends String> changeLogResourceNames = this.getChangeLogResourceNames();
567    if (changeLogResourceNames == null) {
568      returnValue = false;
569    } else {
570      boolean foundOne = false;
571      for (final String changeLogResourceName : changeLogResourceNames) {
572        if (changeLogResourceName != null) {
573          foundOne = this.changeLogResourceExists(changeLogResourceName);
574          if (foundOne) {
575            break;
576          }
577        }
578      }
579      returnValue = foundOne;
580    }
581    this.logger.debug("Exiting changeLogResourceExists(); returning: " + returnValue);
582    return returnValue;
583  }
584
585  protected boolean changeLogResourceExists(final String changeLogResourceName) {
586    this.logger.debug("Entering changeLogResourceExists(String)");
587    final boolean returnValue;
588    if (changeLogResourceName == null) {
589      returnValue = false;
590    } else {
591      ClassLoader cl = Thread.currentThread().getContextClassLoader();
592      if (cl == null) {
593        cl = ClassLoader.getSystemClassLoader();
594        if (cl == null) {
595          cl = this.getClass().getClassLoader();
596        }
597      }
598      returnValue = cl != null && cl.getResource(changeLogResourceName) != null;
599    }
600    this.logger.debug("Exiting changeLogResourceExists(String); returning: " + returnValue);
601    return returnValue;
602  }

655   *
656   * <p>This implementation returns {@code true} if the {@code
657   * liquibase.should.run} system property is non-{@code null} and set
658   * to something other than {@code false} and if the supplied {@link
659   * Liquibase} instance is non-{@code null} and {@linkplain
660   * StandardChangeLogHistoryService#hasDatabaseChangeLogTable() is
661   * connected to a database that does not yet have a
662   * <code>DATABASECHANGELOG</code> table} in it.</p>
663   *
664   * @param liquibase the {@link Liquibase} instance to check; may be
665   * {@code null} in which case {@code false} will be returned
666   *
667   * @return {@code true} if the supplied {@link Liquibase} instance
668   * should in fact have its {@link Liquibase#update(String)} method
669   * called
670   *
671   * @exception LiquibaseException if there was a Liquibase-related
672   * error
673   *
674   * @exception SQLException if there was a database-related error
675   *
676   * @see Liquibase#update(String)
677   */
678  protected boolean shouldUpdate(final Liquibase liquibase) throws LiquibaseException, SQLException {
679    this.logger.debug("Entering shouldUpdate(Liquibase); parameters: liquibase = " + liquibase);
680    boolean returnValue = liquibase != null && !"false".equals(System.getProperty("liquibase.should.run"));
681    if (returnValue) {
682      final Database database = liquibase.getDatabase();
683      if (database != null) {
684        final ChangeLogHistoryService changeLogHistoryService = ChangeLogHistoryServiceFactory.getInstance().getChangeLogService(database);
685        returnValue = !(changeLogHistoryService instanceof StandardChangeLogHistoryService) || !((StandardChangeLogHistoryService)changeLogHistoryService).hasDatabaseChangeLogTable();
686      }
687    }
688    this.logger.debug("Exiting shouldUpdate(Liquibase); returning: " + returnValue);
689    return returnValue;
690  }

692  /**
693   * Creates an {@link AugmentableDatabaseChangeLog} that can
694   * {@linkplain AugmentableDatabaseChangeLog#include(String,
695   * ResourceAccessor)} include other changelogs.
696   *
697   * <p>This method is a hack that exists only because the {@link
698   * DatabaseChangeLog#include(String, boolean, ResourceAccessor}
699   * method is not {@code public}.</p>
700   *
701   * @param database the {@link Database} to use; may be {@code null}
702   *
703   * @param changeLogResourceName the name of the changelog to parse;
704   * may be {@code null}
705   *
706   * @return a new, non-{@code null} {@link
707   * AugmentableDatabaseChangeLog}
708   *
709   * @exception LiquibaseException if an error occurs
710   *
711   * @see DatabaseChangeLog
712   *
713   * @see DatabaseChangeLog#include(String, boolean, ResourceAccessor)
714   */
715  private final AugmentableDatabaseChangeLog createAugmentableDatabaseChangeLog(final Database database, final String changeLogResourceName) throws LiquibaseException {
716    final Liquibase throwaway = new Liquibase(changeLogResourceName, this.getResourceAccessor(), database);
717    final AugmentableDatabaseChangeLog returnValue = new AugmentableDatabaseChangeLog();
718    copyState(throwaway.getDatabaseChangeLog(), returnValue);
719    return returnValue;
720  }

751  /**
752   * A hackish method that would not be needed except that by default
753   * the {@link DatabaseChangeLog#include(String, boolean,
754   * ResourceAccessor)} method is not {@code public}.
755   *
756   * <p>This method copies the state from the {@code oldLog} parameter
757   * into the {@code newLog} parameter.</p>
758   *
759   * <p>The properties that are copied are those that are present in
760   * version 3.3.0 of the {@link DatabaseChangeLog} class.  Updates to
761   * <a href="http://www.liquibase.org/">Liquibase</a> may alter the
762   * state that needs to be copied; in such a case this method will
763   * need to be changed.</p>
764   *
765   * @param oldLog the {@link DatabaseChangeLog} from which state
766   * should be copied; may be {@code null} in which case no action
767   * will be taken
768   *
769   * @param newLog the {@link DatabaseChangeLog} to which state should
770   * be copied; may be {@code null} in which case no action will be
771   * taken
772   *
773   * @see <a
774   * href="https://liquibase.jira.com/browse/CORE-2125"><code>CORE-2125</code></a>
775   */
776  private static final void copyState(final DatabaseChangeLog oldLog, final DatabaseChangeLog newLog) {
777    if (oldLog != null && newLog != null) {
778      newLog.setChangeLogParameters(oldLog.getChangeLogParameters());
779      newLog.setIgnoreClasspathPrefix(oldLog.ignoreClasspathPrefix());
780      newLog.setLogicalFilePath(oldLog.getLogicalFilePath());
781      newLog.setObjectQuotingStrategy(oldLog.getObjectQuotingStrategy());
782      newLog.setPhysicalFilePath(oldLog.getPhysicalFilePath());
783      newLog.setPreconditions(oldLog.getPreconditions());
784      newLog.setRuntimeEnvironment(oldLog.getRuntimeEnvironment());
785      final Collection<ChangeSet> changeSets = oldLog.getChangeSets();
786      assert changeSets != null;
787      for (final ChangeSet changeSet : changeSets) {
788        newLog.addChangeSet(changeSet);
789      }
790    }
791  }
