164      /**
165       * Creates a new {@link LiquiunitRule}.
166       *
167       * <p>The new {@link LiquiunitRule} will have a {@linkplain
168       * #getChangeLogResourceName changelog resource name} of "{@code
169       * changelog.xml}", and a {@link #getResourceAccessor()
170       * ResourceAccessor} {@linkplain
171       * #setResourceAccessor(ResourceAccessor) initialized} with the
172       * following code:</p>
173       *
174       * <!-- Formatting here is important for proper Javadoc output -->
175       * <blockquote><pre>new CompositeResourceAccessor(new URLResourceAccessor(Thread.currentThread().getContextClassLoader()), 
176       *                              new FileSystemResourceAccessor(System.getProperty("user.dir")))</pre></blockquote>
177       *
178       * <p>Consider passing an {@link H2Rule} as the value for the {@code
179       * dataSource} parameter for thread-safe parallel in-memory database
180       * testing scenarios.</p>
181       *
182       * @param dataSource the {@link DataSource} to use to {@linkplain
183       * DataSource#getConnection(String, String) acquire a
184       * <code>Connection</code>}; may be {@code null} in which case this
185       * {@link LiquiunitRule} will effectively do nothing
186       *
187       * @param contexts a variable number of Liquibase <a
188       * href="http://www.liquibase.org/documentation/contexts.html">contexts</a>;
189       * may be {@code null}
190       *
191       * @see DataSource#getConnection(String, String)
192       *
193       * @see #setContexts(Iterable)
194       *
195       * @see #setResourceAccessor(ResourceAccessor)
196       *
197       * @see #setChangeLogResourceName(String)
198       */
199      public LiquiunitRule(final DataSource dataSource, final String... contexts) {
200        super();
201        this.logger = LogFactory.getInstance().getLog("liquiunit");
202        assert this.logger != null;
203        this.logger.debug("Entering LiquiunitRule(DataSource, String[]); parameters: dataSource = " + dataSource + "; contexts = " + Arrays.asList(contexts));
204        this.dataSource = dataSource;
205        this.setChangeLogResourceName("changelog.xml");
206        this.setResourceAccessor(new CompositeResourceAccessor(new URLResourceAccessor(Thread.currentThread().getContextClassLoader()), new FileSystemResourceAccessor(System.getProperty("user.dir"))));
207        if (contexts != null && contexts.length > 0) {
208          this.setContexts(Arrays.asList(contexts));
209        }
210        this.logger.debug("Exiting LiquiunitRule(DataSource, String[])");
211      }

312      /**
313       * Transforms the return value of the {@link #getContexts()} into a
314       * comma-separated {@link String} suitable for passing to the {@link
315       * Liquibase#update(String)} method and returns it.
316       *
317       * @return a comma-separated {@link String} of Liquibase <a
318       * href="http://www.liquibase.org/documentation/contexts.html">contexts</a>,
319       * or {@code null}
320       *
321       * @see #getContexts()
322       *
323       * @see #setContexts(Iterable)
324       *
325       * @see Liquibase#update(String)
326       */
327      private final String getContextsString() {
328        this.logger.debug("Entering getContextsString()");
329        final StringBuilder sb;
330        final Iterable<? extends String> contexts = this.getContexts();
331        if (contexts == null) {
332          sb = null;
333        } else {        
334          final Iterator<? extends String> iterator = contexts.iterator();
335          if (iterator == null) {
336            sb = null;
337          } else {
338            sb = new StringBuilder();
339            while (iterator.hasNext()) {
340              final String context = iterator.next();
341              if (context != null) {
342                sb.append(context);
343                if (iterator.hasNext()) {
344                  sb.append(",");
345                }
346              }
347            }
348          }
349        }
350    
351        final String returnValue;
352        if (sb == null) {
353          returnValue = null;
354        } else {
355          returnValue = sb.toString();
356        }
357        this.logger.debug("Exiting getContextsString(); returning: " + returnValue);
358        return returnValue;
359      }

361      /**
362       * Overrides the {@link ExternalResource#before()} method to
363       * {@linkplain DataSource#getConnection(String, String) acquire a
364       * <code>Connection</code>}, {@linkplain
365       * DatabaseFactory#findCorrectDatabaseImplementation(DatabaseConnection)
366       * find the proper Liquibase <code>Database</code> implementation},
367       * {@linkplain #createLiquibase(Database) create a
368       * <code>Liquibase</code> instance} and, using it, {@linkplain
369       * Liquibase#update(String) update} the backing database using the
370       * {@linkplain #getChangeLogResourceName() affiliated changelog}.
371       *
372       * @exception LiquibaseException if there was a Liquibase-related
373       * error
374       *
375       * @exception SQLException if there was a database-related error
376       * 
377       * @see DataSource#getConnection(String, String)
378       *
379       * @see #createLiquibase(Database)
380       *
381       * @see #shouldUpdate(Liquibase)
382       *
383       * @see #getContexts()
384       *
385       * @see Liquibase#update(String)
386       */
387      @Override
388      protected void before() throws LiquibaseException, SQLException {
389        this.logger.debug("Entering before()");
390        if (this.dataSource != null) {
391    
392          final Connection c = this.dataSource.getConnection();
393          if (c == null) {
394            throw new IllegalStateException("this.dataSource.getConnection()", new NullPointerException("this.dataSource.getConnection()"));
395          } else if (!c.isValid(0)) {
396            throw new IllegalStateException("!c.isValid()");
397          }
398          final JdbcConnection jc = new JdbcConnection(c);
399          try {
400            final DatabaseFactory databaseFactory = DatabaseFactory.getInstance();
401            assert databaseFactory != null;
402            final Database database = databaseFactory.findCorrectDatabaseImplementation(jc);
403            
404            this.liquibase = this.createLiquibase(database);
405            if (this.liquibase != null && this.shouldUpdate(liquibase)) {
406              try {
407                this.liquibase.update(this.getContextsString());
408              } finally {
409                this.liquibase.forceReleaseLocks();
410              }
411            }
412          } finally {
413            try {
414              jc.close();
415            } catch (final DatabaseException ignore) {
416              
417            }
418          }
419    
420        }
421        this.logger.debug("Exiting before()");
422      }

424      /**
425       * Tries to {@linkplain ClassLoader#getResource(String) load} the
426       * {@linkplain #getChangeLogResourceName() classpath resource
427       * identifying a Liquibase changelog}, and, if successful, creates a
428       * new {@link Liquibase} instance initialized to read from it, and
429       * returns it.
430       *
431       * <p>This method may return {@code null}.</p>
432       *
433       * @param database the {@link Database} to operate on; must not be
434       * {@code null}
435       *
436       * @return a new configured {@link Liquibase} instance, or {@code
437       * null} if there is no changelog to read and hence no reason to
438       * construct a {@link Liquibase} instance
439       *
440       * @exception LiquibaseException if a problem occurred while
441       * constructing the new {@link Liquibase} instance
442       *
443       * @see #changeLogResourceExists()
444       *
445       * @see #getResourceAccessor()
446       */
447      protected Liquibase createLiquibase(final Database database) throws LiquibaseException {
448        this.logger.debug("Entering createLiquibase(Database); parameters: database = " + database);
449        final Liquibase liquibase;
450        if (!changeLogResourceExists()) {
451          liquibase = null;
452        } else {
453          liquibase = new Liquibase(changeLogResourceName, this.getResourceAccessor(), database);
454        }
455        this.logger.debug("Exiting createLiquibase(Database); returning: " + liquibase);
456        return liquibase;
457      }

459      /**
460       * Returns {@code true} if the <a
461       * href="http://www.liquibase.org/documentation/databasechangelog.html">changelog</a>
462       * identified by the return value of the {@link
463       * #getChangeLogResourceName()} method actually exists.
464       *
465       * @return {@code true} if the <a
466       * href="http://www.liquibase.org/documentation/databasechangelog.html">changelog</a>
467       * identified by the return value of the {@link
468       * #getChangeLogResourceName()} method actually exists; {@code
469       * false} otherwise
470       *
471       * @see #getChangeLogResourceName()
472       */
473      protected boolean changeLogResourceExists() {
474        this.logger.debug("Entering changeLogResourceExists()");
475        final boolean returnValue;
476        final String changeLogResourceName = this.getChangeLogResourceName();
477        if (changeLogResourceName == null) {
478          returnValue = false;
479        } else {
480          ClassLoader cl = Thread.currentThread().getContextClassLoader();
481          if (cl == null) {
482            cl = ClassLoader.getSystemClassLoader();
483            if (cl == null) {
484              cl = this.getClass().getClassLoader();
485            }
486          }
487          returnValue = cl != null && cl.getResource(changeLogResourceName) != null;
488        }
489        this.logger.debug("Exiting changeLogResourceExists(); returning: " + returnValue);
490        return returnValue;
491      }

538      /**
539       * Returns {@code true} if the supplied {@link Liquibase} instance
540       * should in fact have its {@link Liquibase#update(String)} method
541       * called.
542       *
543       * <p>This implementation returns {@code true} if the {@link
544       * Liquibase#SHOULD_RUN_SYSTEM_PROPERTY} system property is
545       * non-{@code null} and set to something other than {@code false}
546       * and if the supplied {@link Liquibase} instance is non-{@code
547       * null} and {@linkplain
548       * StandardChangeLogHistoryService#hasDatabaseChangeLogTable() is
549       * connected to a database that does not yet have a
550       * <code>DATABASECHANGELOG</code> table} in it.</p>
551       *
552       * @param liquibase the {@link Liquibase} instance to check; may be
553       * {@code null} in which case {@code false} will be returned
554       *
555       * @return {@code true} if the supplied {@link Liquibase} instance
556       * should in fact have its {@link Liquibase#update(String)} method
557       * called
558       *
559       * @exception LiquibaseException if there was a Liquibase-related
560       * error
561       *
562       * @exception SQLException if there was a database-related error
563       *
564       * @see Liquibase#update(String)
565       */
566      protected boolean shouldUpdate(final Liquibase liquibase) throws LiquibaseException, SQLException {
567        this.logger.debug("Entering shouldUpdate(Liquibase); parameters: liquibase = " + liquibase);
568        boolean returnValue = liquibase != null && !"false".equals(System.getProperty(Liquibase.SHOULD_RUN_SYSTEM_PROPERTY));
569        if (returnValue) {
570          final Database database = liquibase.getDatabase();
571          if (database != null) {
572            final ChangeLogHistoryService changeLogHistoryService = ChangeLogHistoryServiceFactory.getInstance().getChangeLogService(database);
573            returnValue = !(changeLogHistoryService instanceof StandardChangeLogHistoryService) || !((StandardChangeLogHistoryService)changeLogHistoryService).hasDatabaseChangeLogTable();
574          }
575        }
576        this.logger.debug("Exiting shouldUpdate(Liquibase); returning: " + returnValue);
577        return returnValue;
578      }
