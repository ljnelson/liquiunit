301      /**
302       * {@linkplain #getConnection(String, String) Opens a new
303       * <code>Connection</code>} to a thread-private H2 in-memory
304       * database and restores any state that was stored by the {@link
305       * #after()} method.
306       *
307       * @see #getConnection(String, String)
308       *
309       * @see H2Archive#loadUnlessEmpty(Description, Connection)
310       *
311       * @see #after()
312       *
313       * @see ExternalResource#before()
314       *
315       * @exception SQLException if a database error occurs
316       *
317       * @exception IllegalStateException if the {@link
318       * #getConnection(String, String)} method returns a {@code null}
319       * {@link Connection} or if the {@link
320       * #configureConnection(Connection)} method causes the {@link
321       * Connection} to become {@linkplain Connection#isValid(int)
322       * invalid}
323       */
324      @Override
325      protected void before() throws SQLException {
326        this.c = this.getConnection(this.username, this.password);
327        if (this.c == null) {
328          throw new IllegalStateException("this.getConnection()", new NullPointerException("this.getConnection()"));
329        }
330        this.configureConnection(this.c);
331        if (!this.c.isValid(0)) {
332          throw new IllegalStateException("this.getConnection().isValid(0)");
333        }
334        if (this.archive != null) {
335          this.archive.loadUnlessEmpty(this.description, this.c);
336        }
337      }

353      private final void setDbCloseDelay() throws SQLException {
354        if (this.c == null) {
355          throw new IllegalStateException("this.getConnection()", new NullPointerException("this.getConnection()"));
356        }
357        final java.sql.Statement statement = this.c.createStatement();
358        assert statement != null;
359        ResultSet rs = null;
360        try {
361          rs = statement.executeQuery("SELECT VALUE FROM INFORMATION_SCHEMA.SETTINGS WHERE NAME = 'DB_CLOSE_DELAY'");
362          assert rs != null;
363          Integer delay = null;
364          if (rs.next()) {
365            delay = rs.getInt(1);
366          }
367          if (rs.wasNull() || delay == null) {
368            rs.close();
369            statement.execute("SET DB_CLOSE_DELAY=-1");
370          }
371        } finally {
372          if (rs != null) {
373            try {
374              rs.close();
375            } catch (final SQLException ignore) {
376    
377            }
378          }
379          try {
380            statement.close();
381          } catch (final SQLException ignore) {
382    
383          }
384        }
385    
386      }

388      /**
389       * Ensures that the current H2 database is {@linkplain
390       * H2Archive#saveIfEmpty(Description, Connection) backed up} and <a
391       * href="http://www.h2database.com/html/grammar.html?highlight=shutdown&search=shutdown#shutdown">shut
392       * down properly</a>.
393       *
394       * @see #before()
395       *
396       * @see ExternalResource#after()
397       */
398      @Override
399      protected void after() {
400        if (this.c != null) {
401    
402          try {
403    
404            if (this.archive != null) {
405              try {
406                this.archive.saveIfEmpty(this.description, this.c);
407              } catch (final SQLException oops) {
408                throw new IllegalStateException(oops);
409              }
410            }
411    
412            java.sql.Statement s = null;
413            try {
414              s = this.c.createStatement();
415              assert s != null;
416              s.execute("SHUTDOWN");
417            } catch (final SQLException shutdownProblem) {
418              throw new IllegalStateException(shutdownProblem);
419            } finally {
420              try {
421                if (s != null) {
422                  s.close();
423                }
424              } catch (final SQLException ignore) {
425    
426              }
427            }
428    
429          } finally {
430            try {
431              this.c.close();
432            } catch (final SQLException ignore) {
433    
434            }
435          }
436    
437          this.c = null;
438        }
439        this.description = null; // XXX TODO INVESTIGATE: not sure this is proper
440      }

518      /**
519       * Internally uses the {@link DriverManager#getConnection(String,
520       * String, String)} method to create a {@link Connection} to a new,
521       * in-memory H2 database that is <a
522       * href="https://groups.google.com/d/msg/h2-database/FjteyQPoiGg/IffNPLQjirAJ"
523       * target="_parent">safe for use</a> by the current {@link Thread},
524       * not just the current JVM.
525       *
526       * <p>In-memory H2 databases by default are private to a given
527       * {@link Connection}.  For most unit- and integration-testing
528       * purposes, that is not sufficient, as often several {@link
529       * Connection}s have to be opened to those databases to perform
530       * various operations.  What you want is a way to have two
531       * independently created {@link Connection}s connect to the same
532       * in-memory H2 database instance.</p>
533       *
534       * <p>This is easy to do, of course&mdash;you simply name the
535       * database (e.g. {@code jdbc:h2:mem:fred} instead of just {@code
536       * jdbc:h2:mem:}).  But you want to isolate this name from other
537       * threads and processes that might try to connect to it, so you
538       * need to pick a name that is unique to the current {@link Thread}.
539       * This method accomplishes just such a {@link Connection}.</p>
540       *
541       * <p>The JDBC URL that is used to open a {@link Connection} is
542       * built according to the following template:</p>
543       *
544       * <blockquote><pre>jdbc:h2:mem:[TEST_NAME-]pid=PID-thread=THREAD_ID[;INIT=INIT_SQL]</pre></blockquote>
545       *
546       * <p>...where brackets denote optional elements, {@code TEST_NAME}
547       * is the return value of the {@link Description#getDisplayName()}
548       * method, {@code PID} is the return value of the {@link #pid()}
549       * method, {@code THREAD_ID} is the return value of the {@link
550       * Thread#getId()} method when invoked on the {@linkplain
551       * Thread#currentThread() current <code>Thread</code>} and {@code
552       * INIT_SQL} is the initialization SQL passed to {@linkplain
553       * #H2Rule(String, String, String) the constructor}.</p>
554       *
555       * <p>The {@link Connection} returned is guaranteed to be non-{@code
556       * null} and not {@linkplain Connection#isClosed() closed}.</p>
557       *
558       * @param username the username to use; may be {@code null}
559       *
560       * @param password the password to use; may be {@code null}
561       *
562       * @return a non-{@code null} open {@link Connection}
563       *
564       * @exception SQLException if an error occurs
565       *
566       * @see DriverManager#getConnection(String, String, String)
567       *
568       * @see DriverManager#getConnection(String)
569       *
570       * @see <a
571       * href="http://h2database.com/html/features.html#in_memory_databases">The
572       * section on in-memory databases in the H2 documentation</a>
573       */
574      @Override
575      public Connection getConnection(final String username, final String password) throws SQLException {
576        final String url = this.getConnectionURL();
577        if (username != null) {
578          return DriverManager.getConnection(url, username, password);
579        } else {
580          return DriverManager.getConnection(url);
581        }
582      }

584      public String getConnectionURL() {
585        final StringBuilder sb = new StringBuilder("jdbc:h2:mem:");
586        if (this.description != null) {
587          final String displayName = this.description.getDisplayName();
588          if (displayName != null) {
589            sb.append(displayName);
590            sb.append("-");
591          }
592        }
593        sb.append("pid=");
594        sb.append(this.pid());
595        sb.append("-thread=");
596        sb.append(Thread.currentThread().getId());
597        if (this.initSql != null) {
598          final String sql = this.initSql.trim();
599          if (sql != null && !sql.isEmpty()) {
600            sb.append(";INIT=");
601            sb.append(sql);
602          }
603        }
604        return sb.toString();
605      }

633      /**
634       * Returns the current process identifier as a {@link String}.
635       *
636       * <p>This method may return {@code null} in exceptional
637       * circumstances.</p>
638       *
639       * <p>The default implementation returns a {@link String} resulting
640       * from the following invocation:</p>
641       *
642       * <blockquote><pre>java.lang.management.ManagementFactory.getRuntimeMXBean().getName().substring(0, name.indexOf('@'));</pre></blockquote>
643       *
644       * @return the current process identifier as a {@link String}, or
645       * {@code null}
646       *
647       * @see java.lang.management.ManagementFactory#getRuntimeMXBean()
648       *
649       * @see java.lang.management.RuntimeMXBean#getName()
650       */
651      public String pid() {
652        final String pid;
653        final String name = java.lang.management.ManagementFactory.getRuntimeMXBean().getName();
654        if (name == null) {
655          pid = null;
656        } else {
657          pid = name.substring(0, name.indexOf('@'));
658        }
659        return pid;
660      }
