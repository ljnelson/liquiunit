064/**
065 * An {@link ExternalResource} that wraps JUnit tests with a {@link

233    }
234    this.tester = tester;

243  /**
244   * Ensures that every {@link IDatabaseConnection} produced during
245   * the course of execution is configured to use the supplied {@link
246   * IDataTypeFactory}.
247   *
248   * @param dataTypeFactory the {@link IDataTypeFactory} suitable for
249   * the underlying database; may be {@code null}
250   *
251   * @see IOperationListener#connectionRetrieved(IDatabaseConnection)
252   *
253   * @see DatabaseConfig#PROPERTY_DATATYPE_FACTORY
254   */
255  private final void installDataTypeFactory(final IDataTypeFactory dataTypeFactory) {
256    if (dataTypeFactory != null && this.tester != null) {
257      this.tester.setOperationListener(new DefaultOperationListener() {
258          @Override
259          public final void connectionRetrieved(final IDatabaseConnection connection) {
260            super.connectionRetrieved(connection);
261            if (connection != null) {
262              final DatabaseConfig config = connection.getConfig();
263              if (config != null) {
264                config.setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, dataTypeFactory);
265              }
266            }
267          }
268        });
269    }
270  }

272  /**
273   * {@linkplain #getDataSet(Description) Finds an appropriate
274   * <code>IDataSet</code>} for the {@linkplain #tester affiliated
275   * <code>DataSourceDatabaseTester</code>} and {@linkplain
276   * AbstractDatabaseTester#setDataSet(IDataSet) installs it}
277   * immediately before invoking the {@link IDatabaseTester#onSetup()}
278   * method.
279   *
280   * @exception Exception if an error occurs
281   */
282  @Override
283  public void before() throws Exception {
284    if (this.tester != null) {
285      final IDataSet oldDataSet = this.tester.getDataSet();
286      this.oldDataSet = oldDataSet;
287      if (oldDataSet == null) {
288        IDataSet newDataSet = this.getDataSet(this.description);
289        if (newDataSet == null) {
290          newDataSet = new DefaultDataSet();
291        }
292        this.tester.setDataSet(newDataSet);
293      }
294      this.tester.onSetup();
295    }
296  }
297
298  /**
299   * Invokes the {@link IDatabaseTester#onTearDown()} method.
300   *
301   * @see IDatabaseTester#onTearDown()
302   */
303  @Override
304  public void after() {
305    if (this.tester != null) {
306      try {
307        this.tester.onTearDown();
308      } catch (final RuntimeException throwMe) {
309        throw throwMe;
310      } catch (final Exception everythingElse) {
311        throw new RuntimeException(everythingElse); // TODO: ugly
312      } finally {
313        try {
314          this.closeDataSet();
315        } catch (final IOException ignore) {
316
317        }
318      }
319      this.tester.setDataSet(this.oldDataSet);
320    }
321    this.description = null; // XXX TODO INVESTIGATE: not sure this is proper
322  }

354  /**
355   * Follows various conventions, detailed below, in attempting to
356   * locate and instantiate a new {@link IDataSet} instance
357   * appropriate for the supplied {@link Description}.
358   *
359   * <p>This implementation first checks the {@link #tester
360   * DataSourceDatabaseTester} indirectly or directly supplied to this
361   * {@link DataSourceDatabaseTesterRule} at {@linkplain
362   * #DataSourceDatabaseTesterRule(DataSource) construction time} to
363   * see if {@linkplain DataSourceDatabaseTester#getDataSet() it
364   * already has a <code>IDataSet</code> implementation installed}.
365   * If so, then no further action is taken and that {@link IDataSet}
366   * is returned.  ({@code null} is returned if the supplied {@code
367   * description} is {@code null}.)</p>
368   *
369   * <p>Otherwise, a {@linkplain ClassLoader#getResource(String)
370   * classpath resource} named {@code
371   * datasets/SIMPLE_TEST_CLASS_NAME/TEST_METHOD_NAME.xml} is sought
372   * using the {@linkplain Thread#getContextClassLoader() context
373   * classloader}, where {@code SIMPLE_TEST_CLASS_NAME} is the name of
374   * the JUnit test {@link Class} currently running and {@code
375   * TEST_METHOD_NAME} is the name of the JUnit test method currently
376   * running.</p>
377   *
378   * <p>If that resource doesn't exist, then a {@linkplain
379   * ClassLoader#getResource(String) classpath resource} named {@code
380   * datasets/SIMPLE_TEST_CLASS_NAME.xml} is sought using the
381   * {@linkplain Thread#getContextClassLoader() context classloader},
382   * where {@code SIMPLE_TEST_CLASS_NAME} is the name of the JUnit
383   * test {@link Class} currently running.</p>
384   *
385   * <p>Once a resource is located in this manner, its {@link URL} is
386   * passed to the {@link #createDataSet(URL)} method, and that
387   * method's return value is returned.</p>
388   *
389   * <p>If no resource exists, then {@code null} is returned.</p>
390   *
391   * @param description a {@link Description} describing the JUnit
392   * test being executed; may be {@code null}
393   *
394   * @return an {@link IDataSet} instance appropriate for the supplied
395   * {@link Description}, or {@code null}
396   *
397   * @exception DataSetException if there was an error in constructing
398   * the {@link IDataSet}
399   *
400   * @exception IOException if there was an input/output error
401   *
402   * @see #createDataSet(URL)
403   */
404  protected IDataSet getDataSet(final Description description) throws DataSetException, IOException {
405    final IDataSet returnValue;
406    if (this.tester == null) {
407      returnValue = null;
408    } else {
409      final IDataSet old = this.tester.getDataSet();
410      if (old == null && description != null) {
411        final String simpleClassName;
412        final Class<?> testClass = description.getTestClass();
413        if (testClass == null) {
414          simpleClassName = null;
415        } else {
416          simpleClassName = testClass.getSimpleName();
417        }
418        final String methodName = description.getMethodName();
419        ClassLoader cl = Thread.currentThread().getContextClassLoader();
420        if (cl == null) {
421          cl = ClassLoader.getSystemClassLoader();
422          if (cl == null) {
423            cl = this.getClass().getClassLoader();
424          }
425        }
426        assert cl != null;
427        URL url = cl.getResource(String.format("datasets/%s/%s.xml", simpleClassName, methodName));
428        if (url == null) {
429          url = cl.getResource(String.format("datasets/%s.xml", simpleClassName));
430        }
431        returnValue = this.createDataSet(url);
432      } else {
433        returnValue = old;
434      }
435    }
436    return returnValue;
437  }

439  /**
440   * Creates a new {@link IDataSet} implementation suitable for the
441   * supplied {@link URL} and returns it.
442   *
443   * <p>This method may return {@code null}.</p>
444   *
445   * <p>Overrides of this method are permitted to return {@code
446   * null}.</p>
447   *
448   * <p>This implementation returns {@code null} if a {@code null}
449   * {@link URL} is supplied.  Otherwise, a new {@link XmlDataSet} is
450   * constructed with the {@linkplain URL#openStream() supplied
451   * <code>URL</code>'s affiliated <code>InputStream</code>} and
452   * returned.</p>
453   *
454   * @param url the {@link URL} for which a new {@link IDataSet}
455   * implementation should be returned; may be {@code null}
456   *
457   * @return a new {@link IDataSet} implementation, or {@code null}
458   *
459   * @exception DataSetException if an error occurs during {@link
460   * IDataSet} construction
461   *
462   * @exception IOException if an error occurs during the processing
463   * of the supplied {@link URL}
464   *
465   * @see #getDataSet(Description)
466   */
467  protected IDataSet createDataSet(final URL url) throws DataSetException, IOException {
468    final IDataSet returnValue;
469    if (url == null) {
470      returnValue = null;
471    } else {
472      final InputStream stream = url.openStream();
473      this.dataSetInputStream = stream;
474      returnValue = new XmlDataSet(stream);
475    }
476    return returnValue;
477  }

479  /**
480   * Notionally closes any resources held by the {@link IDataSet}
481   * created by the {@link #createDataSet(URL)} method.
482   *
483   * @exception IOException if an error occurs during closing
484   *
485   * @see #createDataSet(URL)
486   */
487  protected void closeDataSet() throws IOException {
488    if (this.dataSetInputStream != null) {
489      this.dataSetInputStream.close();
490    }
491  }
