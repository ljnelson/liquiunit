054/**
055 * An {@link ExternalResource} that sets up an {@link EntityManager}
056 * before each JUnit test run.
057 *
058 * <p>Clients may choose whether a single {@link EntityManagerFactory}
059 * is used to create all {@link EntityManager}s, or whether an {@link
060 * EntityManagerFactory} is created and {@linkplain
061 * EntityManagerFactory#close() closed} before and after each test
062 * run.</p>
063 *
064 * <p>Although creating an {@link EntityManagerFactory} can be a
065 * heavyweight operation, it offers much more isolation than the
066 * alternative, and the intensity of its resource usage can be offset
067 * by running tests in parallel.</p>
068 *
069 * @author <a href="http://about.me/lairdnelson"
070 * target="_parent">Laird Nelson</a>
071 *
072 * @see EntityManagerFactory
073 *
074 * @see ExternalResource
075 *
076 * @see Persistence#createEntityManagerFactory(String)
077 */
078public class JPARule extends ExternalResource {

335  /**
336   * Creates an {@link EntityManager} and injects it into all fields
337   * in the test instance that are of type {@code EntityManager} and
338   * that are annotated with {@link
339   * PersistenceContext @PersistenceContext}.
340   *
341   * @exception IllegalAccessException if there was a reflection
342   * problem
343   *
344   * @exception InvocationTargetException if there was a reflection
345   * problem
346   */
347  @Override
348  public void before() throws IllegalAccessException, InvocationTargetException {
349
350    // If there's no EntityManagerFactory yet, or a pre-existing one,
351    // acquire a new one for this test.
352    if (this.emf == null || !this.emf.isOpen()) {
353      this.closeFactory = true;
354      final Map<?, ?> entityManagerFactoryProperties = this.getEntityManagerFactoryProperties();
355      if (entityManagerFactoryProperties == null || entityManagerFactoryProperties.isEmpty()) {
356        this.emf = Persistence.createEntityManagerFactory(this.persistenceUnitName);
357      } else {
358        this.emf = Persistence.createEntityManagerFactory(this.persistenceUnitName, entityManagerFactoryProperties);
359      }
360    }
361
362    if (this.emf != null && this.emf.isOpen()) {
363
364      // If for some crazy reason there's an open EntityManager
365      // already installed, close it.
366      if (this.em != null && this.em.isOpen()) {
367        this.em.close();
368      }
369
370      // Create a new EntityManager.
371      if (this.entityManagerProperties == null || this.entityManagerProperties.isEmpty()) {
372        this.em = this.emf.createEntityManager();
373      } else {
374        this.em = this.emf.createEntityManager(this.entityManagerProperties);
375      }
376
377      if (this.description != null) {
378        final Class<?> testClass = description.getTestClass();
379        if (testClass != null) {
380
381          // Look for all @PersistenceContext-annotated fields in the
382          // test class and its superclasses.  Set the just-created
383          // EntityManager into all of them on the current test
384          // instance.  Keep track of which fields were set so they
385          // can be cleared in the after() method.
386
387          final Collection<Field> fields = new ArrayList<Field>();
388          Class<?> cls = testClass;
389          while (cls != null) {
390            final Field[] declaredFields = cls.getDeclaredFields();
391            if (declaredFields != null && declaredFields.length > 0) {
392              for (final Field f : declaredFields) {
393                if (f != null) {
394                  if (EntityManager.class.isAssignableFrom(f.getType())) {
395                    final PersistenceContext pc = f.getAnnotation(PersistenceContext.class);
396                    if (pc != null) {
397                      final boolean accessible = f.isAccessible();
398                      try {
399                        if (testClass.equals(f.getDeclaringClass())) {
400                          f.setAccessible(true);
401                        }
402                        f.set(this.testInstance, this.em);
403                        fields.add(f);
404                      } catch (final SecurityException ohWell) {
405                        // ignore
406                      } finally {
407                        f.setAccessible(accessible);
408                      }
409                    }
410                  }
411                }
412              }
413            }
414            cls = cls.getSuperclass();
415          }
416          this.fields = fields;
417        }
418      }
419
420      assert this.em != null;
421      this.et = this.em.getTransaction();
422      if (this.et != null) {
423        this.et.begin();
424      }
425    }
426  }

428  /**
429   * Returns a {@link Map} that will be passed to the {@link
430   * Persistence#createEntityManagerFactory(String, Map)} method to
431   * create a new {@link EntityManagerFactory}.
432   *
433   * <p>Note that if an {@link H2Rule} has been passed to the
434   * {@linkplain #JPARule(Object, String, H2Rule) relevant
435   * constructor}, then the {@link Map} returned will be the return
436   * value of the {@link H2Rule#getJPAProperties()} method.</p>
437   *
438   * <p>This method may return {@code null}.</p>
439   *
440   * @return a {@link Map} of {@link EntityManagerFactory} properties,
441   * or {@code null}
442   *
443   * @see Persistence#createEntityManagerFactory(String, Map)
444   * 
445   * @see H2Rule#getJPAProperties()
446   */
447  public Map<?, ?> getEntityManagerFactoryProperties() {
448    final Map<?, ?> returnValue;
449    if (this.h2Rule != null) {
450      returnValue = this.h2Rule.getJPAProperties();
451    } else {
452      returnValue = this.entityManagerFactoryProperties;
453    }
454    return returnValue;
455  }

458   * Overrides the {@link ExternalResource#apply(Statement,

470   *
471   * @param description the {@link Description} describing the test
472   * underway; the superclass documentation does not define what
473   * behavior will occur if this parameter is {@code null}

478   *
479   * @see ExternalResource#apply(Statement, Description)

487  /**
488   * {@linkplain EntityTransaction#rollback() Rolls back} any
489   * {@linkplain EntityTransaction#isActive() active} {@link
490   * EntityTransaction}, {@linkplain EntityManager#close() closes} any
491   * {@linkplain EntityManager#isOpen() open} {@link EntityManager},
492   * sets all fields that were injected by the {@link #before()}
493   * method to {@code null}, and closes any {@link
494   * EntityManagerFactory} that was created by this {@link JPARule}.
495   *
496   * @see #before()
497   */
498  @Override
499  public void after() {
500    if (this.et != null && this.et.isActive()) {
501      try {
502        this.et.rollback();
503      } catch (final PersistenceException ohWell) {
504        ohWell.printStackTrace();
505      }
506      this.et = null;
507    }
508    if (this.em != null && this.em.isOpen()) {
509      final EntityTransaction et = this.em.getTransaction();
510      if (et != null && et.isActive()) {
511        et.rollback();
512      }
513      this.em.close();      
514    }    
515    this.em = null;
516    if (this.fields != null && !this.fields.isEmpty()) {
517      for (final Field f : this.fields) {
518        if (f != null) {
519          final boolean old = f.isAccessible();
520          try {
521            f.setAccessible(true);
522            f.set(this.testInstance, null);
523          } catch (final IllegalAccessException wrapMe) {
524            throw new RuntimeException(wrapMe);
525          } finally {
526            f.setAccessible(old);
527          }
528        }
529      }
530      this.fields = null;
531    }
532    if (this.closeFactory && this.emf != null && this.emf.isOpen()) {
533      this.emf.close();
534      this.closeFactory = false;
535    }
536    this.description = null;
537  }

539  /**
540   * Forcibly closes any open {@link EntityManager}s and {@link
541   * EntityManagerFactory} instances, thus releasing resources.
542   *
543   * <p>Most callers do not need to invoke this method.</p>
544   */
545  public void close() {
546    if (this.em != null) {
547      this.em.close();
548      this.em = null;
549    }
550    if (this.emf != null) {
551      this.emf.close();
552      this.emf = null;
553    }
554  }

579  /**
580   * A convenience "fire-and-forget" method that returns a {@link
581   * TestRule} constructed in the following manner:
582   *
583   * <blockquote><pre>final H2Rule h2 = new H2Rule(archive);
584   *final LiquiunitRule liquibase = new LiquiunitRule(h2);
585   *final DataSourceDatabaseTesterRule dbUnit = new DataSourceDatabaseTesterRule(h2, new H2DataTypeFactory());
586   *final TestRule jpaRule = new JPARule(this, "test", h2);
587   *return RuleChain.outerRule(h2).around(liquibase).around(dbUnit).around(jpaRule);</pre></blockquote>
588   *
589   * @param testInstance the JUnit test instance being set up; should
590   * not be {@code null}
591   * 
592   * @param archive an {@link H2Archive} to store the "blank" state of
593   * the underlying in-memory H2 database; may be {@code null}
594   * 
595   * @return a non-{@code null} {@link TestRule} set up for
596   * JPA-oriented testing
597   */
598  public static TestRule typical(final Object testInstance, final H2Archive archive) {
599    final H2Rule h2 = new H2Rule(archive);
600    final LiquiunitRule liquibase = new LiquiunitRule(h2);
601    final DataSourceDatabaseTesterRule dbUnit = new DataSourceDatabaseTesterRule(h2, new H2DataTypeFactory());
602    final TestRule jpaRule = new JPARule(testInstance, "test", h2);
603    // H2, then Liquibase, then dbUnit, then JPA.
604    return RuleChain.outerRule(h2).around(liquibase).around(dbUnit).around(jpaRule);
605  }
