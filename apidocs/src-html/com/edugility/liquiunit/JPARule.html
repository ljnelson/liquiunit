049    /**
050     * An {@link ExternalResource} that sets up an {@link EntityManager}
051     * before each JUnit test run.
052     *
053     * <p>Clients may choose whether a single {@link EntityManagerFactory}
054     * is used to create all {@link EntityManager}s, or whether an {@link
055     * EntityManagerFactory} is created and {@linkplain
056     * EntityManagerFactory#close() closed} before and after each test
057     * run.</p>
058     *
059     * <p>Although creating an {@link EntityManagerFactory} can be a
060     * heavyweight operation, it offers much more isolation than the
061     * alternative, and the intensity of its resource usage can be offset
062     * by running tests in parallel.</p>
063     *
064     * @author <a href="http://about.me/lairdnelson"
065     * target="_parent">Laird Nelson</a>
066     *
067     * @see EntityManagerFactory
068     *
069     * @see ExternalResource
070     *
071     * @see Persistence#createEntityManagerFactory(String)
072     */
073    public class JPARule extends ExternalResource {

294       *
295       * @param testInstance an instance of the test class; must not be

310        super();
311        if (testInstance == null) {
312          throw new IllegalArgumentException("testInstance", new NullPointerException("testInstance"));
313        }
314        this.testInstance = testInstance;
315        this.persistenceUnitName = null;
316        this.h2Rule = null;
317        this.entityManagerFactoryProperties = emf == null ? null : emf.getProperties();
318        this.entityManagerProperties = entityManagerProperties;
319        this.emf = emf;
320      }

328      /**
329       * Creates an {@link EntityManager} and injects it into all fields
330       * in the test instance that are of type {@code EntityManager} and
331       * that are annotated with {@link
332       * PersistenceContext @PersistenceContext}.
333       *
334       * @exception IllegalAccessException if there was a reflection
335       * problem
336       *
337       * @exception InvocationTargetException if there was a reflection
338       * problem
339       */
340      @Override
341      public void before() throws IllegalAccessException, InvocationTargetException {
342    
343        // If there's no EntityManagerFactory yet, or a pre-existing one,
344        // acquire a new one for this test.
345        if (this.emf == null || !this.emf.isOpen()) {
346          this.closeFactory = true;
347          final Map<?, ?> entityManagerFactoryProperties = this.getEntityManagerFactoryProperties();
348          if (entityManagerFactoryProperties == null || entityManagerFactoryProperties.isEmpty()) {
349            this.emf = Persistence.createEntityManagerFactory(this.persistenceUnitName);
350          } else {
351            this.emf = Persistence.createEntityManagerFactory(this.persistenceUnitName, entityManagerFactoryProperties);
352          }
353        }
354    
355        if (this.emf != null && this.emf.isOpen()) {
356    
357          // If for some crazy reason there's an open EntityManager
358          // already installed, close it.
359          if (this.em != null && this.em.isOpen()) {
360            this.em.close();
361          }
362    
363          // Create a new EntityManager.
364          if (this.entityManagerProperties == null || this.entityManagerProperties.isEmpty()) {
365            this.em = this.emf.createEntityManager();
366          } else {
367            this.em = this.emf.createEntityManager(this.entityManagerProperties);
368          }
369    
370          if (this.description != null) {
371            final Class<?> testClass = description.getTestClass();
372            if (testClass != null) {
373    
374              // Look for all @PersistenceContext-annotated fields in the
375              // test class and its superclasses.  Set the just-created
376              // EntityManager into all of them on the current test
377              // instance.  Keep track of which fields were set so they
378              // can be cleared in the after() method.
379    
380              final Collection<Field> fields = new ArrayList<Field>();
381              Class<?> cls = testClass;
382              while (cls != null) {
383                final Field[] declaredFields = cls.getDeclaredFields();
384                if (declaredFields != null && declaredFields.length > 0) {
385                  for (final Field f : declaredFields) {
386                    if (f != null && EntityManager.class.isAssignableFrom(f.getType())) {
387                      final PersistenceContext pc = f.getAnnotation(PersistenceContext.class);
388                      if (pc != null) {
389                        final boolean accessible = f.isAccessible();
390                        try {
391                          if (testClass.equals(f.getDeclaringClass())) {
392                            f.setAccessible(true);
393                          }
394                          f.set(this.testInstance, this.em);
395                          fields.add(f);
396                        } catch (final SecurityException ohWell) {
397                          // ignore
398                        } finally {
399                          f.setAccessible(accessible);
400                        }
401                      }
402                    }
403                  }
404                }
405                cls = cls.getSuperclass();
406              }
407              this.fields = fields;
408            }
409          }
410        }
411      }

413      /**
414       * Returns a {@link Map} that will be passed to the {@link
415       * Persistence#createEntityManagerFactory(String, Map)} method to
416       * create a new {@link EntityManagerFactory}.
417       *
418       * <p>Note that if an {@link H2Rule} has been passed to the
419       * {@linkplain #JPARule(Object, String, H2Rule) relevant
420       * constructor}, then the {@link Map} returned will be the return
421       * value of the {@link H2Rule#getJPAProperties()} method.</p>
422       *
423       * <p>This method may return {@code null}.</p>
424       *
425       * @return a {@link Map} of {@link EntityManagerFactory} properties,
426       * or {@code null}
427       *
428       * @see Persistence#createEntityManagerFactory(String, Map)
429       * 
430       * @see H2Rule#getJPAProperties()
431       */
432      public Map<?, ?> getEntityManagerFactoryProperties() {
433        final Map<?, ?> returnValue;
434        if (this.h2Rule != null) {
435          returnValue = this.h2Rule.getJPAProperties();
436        } else {
437          returnValue = this.entityManagerFactoryProperties;
438        }
439        return returnValue;
440      }

442      /**
443       * Overrides the {@link ExternalResource#apply(Statement,
444       * Description)} method to store the supplied {@link Description}
445       * for usage by the {@link #before()} method internally and returns
446       * the superclass' return value.
447       *
448       * <p>It must be assumed that this method may return {@code null}
449       * since the superclass documentation does not mention whether the
450       * return value must be non-{@code null}.</p>
451       *
452       * @param base the {@link Statement} to decorate; the superclass
453       * documentation does not define what behavior will occur if this
454       * parameter is {@code null}
455       *
456       * @param description the {@link Description} describing the test
457       * underway; the superclass documentation does not define what
458       * behavior will occur if this parameter is {@code null}
459       *
460       * @return a {@link Statement}; possibly {@code null}
461       *
462       * @see #before()
463       *
464       * @see ExternalResource#apply(Statement, Description)
465       */
466      @Override
467      public Statement apply(final Statement base, final Description description) {
468        this.description = description;
469        return super.apply(base, description);
470      }

472      /**
473       * {@linkplain EntityTransaction#rollback() Rolls back} any
474       * {@linkplain EntityTransaction#isActive() active} {@link
475       * EntityTransaction}, {@linkplain EntityManager#close() closes} any
476       * {@linkplain EntityManager#isOpen() open} {@link EntityManager},
477       * sets all fields that were injected by the {@link #before()}
478       * method to {@code null}, and closes any {@link
479       * EntityManagerFactory} that was created by this {@link JPARule}.
480       *
481       * @see #before()
482       */
483      @Override
484      public void after() {
485        if (this.em != null && this.em.isOpen()) {
486          final EntityTransaction et = this.em.getTransaction();
487          if (et != null && et.isActive()) {
488            et.rollback();
489          }
490          this.em.close();      
491        }    
492        this.em = null;
493        if (this.fields != null && !this.fields.isEmpty()) {
494          for (final Field f : this.fields) {
495            if (f != null) {
496              final boolean old = f.isAccessible();
497              try {
498                f.setAccessible(true);
499                f.set(this.testInstance, null);
500              } catch (final IllegalAccessException wrapMe) {
501                throw new RuntimeException(wrapMe);
502              } finally {
503                f.setAccessible(old);
504              }
505            }
506          }
507          this.fields = null;
508        }
509        if (this.closeFactory && this.emf != null && this.emf.isOpen()) {
510          this.emf.close();
511          this.closeFactory = false;
512        }
513        this.description = null;
514      }
515    
516      /**
517       * Forcibly closes any open {@link EntityManager}s and {@link
518       * EntityManagerFactory} instances, thus releasing resources.
519       *
520       * <p>Most callers do not need to invoke this method.</p>
521       */
522      public void close() {
523        if (this.em != null) {
524          this.em.close();
525          this.em = null;
526        }
527        if (this.emf != null) {
528          this.emf.close();
529          this.emf = null;
530        }
531      }
