028package com.edugility.liquiunit;
029
030import java.sql.Connection;
031import java.sql.ResultSet;
032import java.sql.Statement;
033import java.sql.SQLException;
034
035import java.util.concurrent.locks.ReadWriteLock;
036import java.util.concurrent.locks.ReentrantReadWriteLock;
037
038import org.junit.runner.Description;
039
040/**
041 * A special-purpose class that saves an in-memory <a
042 * href="http://www.h2database.com/">H2</a> database as a {@link
043 * String} <a
044 * href="http://www.h2database.com/html/grammar.html?#script">containing
045 * all the DML and DDL to reconstitute it</a>.
046 *
047 * <p>This class is designed for use only in light integration-testing
048 * scenarios.</p>
049 *
050 * <p>This class is safe for use by multiple threads.</p>
051 *
052 * @author <a href="http://about.me/lairdnelson"
053 * target="_parent">Laird Nelson</a>
054 *
055 * @see H2Rule
056 *
057 * @see <a href="http://www.h2database.com/html/grammar.html?#script">the <code>SCRIPT</code> command</a>
058 */
059public class H2Archive {
060
061  /**
062   * A {@link ReadWriteLock} used to protect access to the data stored
063   * as an archive of a given H2 database.
064   *
065   * <p>This field is never {@code null}.</p>
066   */
067  protected final ReadWriteLock dataLock;
068
069  /**
070   * The data that comprises the H2 archive.
071   *
072   * <p>This field may be {@code null} at any point.</p>
073   *
074   * <p>Access to this field must be mediated by the {@link #dataLock}
075   * field.</p>
076   *
077   * @see #dataLock
078   */
079  private volatile String data;
080
081  /**
082   * Creates a new {@link H2Archive}.
083   */
084  public H2Archive() {
085    super();
086    this.dataLock = new ReentrantReadWriteLock();
087  }
088
089  /**
090   * Provided that no archive data currently exists, atomically saves
091   * the state of the H2 database reachable via the supplied {@link
092   * Connection} as the newline-separated output of the H2 {@code
093   * SCRIPT} command.
094   *
095   * <p>If this method has been called before, then a subsequent
096   * invocation will not overwrite any existing data.</p>
097   *
098   * @param description the {@link Description} describing the current
099   * JUnit test underway; ignored by this implementation but may be
100   * useful for subclasses; may be {@code null}
101   *
102   * @param c a {@link Connection} to an in-memory H2 database; must
103   * not be {@code null}; must be {@linkplain Connection#isValid(int)
104   * valid}
105   *
106   * @return {@code true} if a save actually occurred; {@code false}
107   * if no action was taken
108   *
109   * @exception IllegalArgumentException if {@code c} is {@code null}
110   *
111   * @exception IllegalStateException if the supplied {@link
112   * Connection} is not {@linkplain Connection#isValid(int) valid}
113   *
114   * @exception SQLException if a database error occurs
115   *
116   * @see #loadUnlessEmpty(Description, Connection)
117   */
118  public boolean saveIfEmpty(final Description description, final Connection c) throws SQLException {
119    if (c == null) {
120      throw new IllegalArgumentException("c", new NullPointerException("c"));
121    } else if (!c.isValid(0)) {
122      throw new IllegalStateException("this.getConnection().isValid(0)");
123    }
124    boolean returnValue = false;
125    
126    try {
127      this.dataLock.writeLock().lock();
128      if (this.isEmpty(description)) {
129        
130        final StringBuilder sb = new StringBuilder();
131        
132        final Statement statement = c.createStatement();
133        assert statement != null;
134        ResultSet rs = null;
135        try {
136          rs = statement.executeQuery("SCRIPT");
137          assert rs != null;
138          while (rs.next()) {
139            sb.append(rs.getString(1)).append("\n");
140          }
141        } finally {
142          if (rs != null) {
143            try {
144              rs.close();
145            } catch (final SQLException neverMind) {
146              // ignore
147            }
148          }
149          try {
150            statement.close();
151          } catch (final SQLException neverMind) {
152            // ignore
153          }
154        }
155        
156        this.data = sb.toString();
157        returnValue = !this.isEmpty(description);
158      }
159    } finally {
160      this.dataLock.writeLock().unlock();
161    }
162    return returnValue;
163  }
164
165  /**
166   * Uses any stored DML or DDL commands this {@link H2Archive}
167   * encapsulates (saved earlier by the {@link
168   * #saveIfEmpty(Description, Connection)} method) to populate the
169   * in-memory H2 database represented and attached to by the supplied
170   * {@link Connection}.
171   *
172   * @param description the {@link Description} describing the current
173   * JUnit test underway; ignored by this implementation but may be
174   * useful for subclasses; may be {@code null}
175   *
176   * @param c a {@link Connection} to an in-memory H2 database; must
177   * not be {@code null}; must be {@linkplain Connection#isValid(int)
178   * valid}
179   *
180   * @return {@code true} if the state of the database was altered;
181   * {@code false} otherwise
182   * 
183   * @exception IllegalArgumentException if {@code c} is {@code null}
184   *
185   * @exception IllegalStateException if the supplied {@link
186   * Connection} is not {@linkplain Connection#isValid(int) valid}
187   *
188   * @exception SQLException if a database error occurs
189   *
190   * @see #saveIfEmpty(Description, Connection)
191   */
192  public boolean loadUnlessEmpty(final Description description, final Connection c) throws SQLException {
193    if (c == null) {
194      throw new IllegalStateException("this.getConnection()", new NullPointerException("this.getConnection()"));
195    } else if (!c.isValid(0)) {
196      throw new IllegalStateException("this.getConnection().isValid(0)");
197    }
198    boolean returnValue = false;
199    try {
200      this.dataLock.readLock().lock();
201      if (!this.isEmpty(description)) {
202        final Statement statement = c.createStatement();
203        assert statement != null;
204        try {
205          statement.execute(this.data);
206        } finally {
207          try {
208            statement.close();
209          } catch (final SQLException neverMind) {
210            // ignore on purpose
211          }
212        }
213        returnValue = true;
214      }
215    } finally {
216      this.dataLock.readLock().unlock();
217    }
218    return returnValue;
219  }
220
221  /**
222   * Returns {@code true} if, for the supplied {@link Description},
223   * this {@link H2Archive} is conceptually
224   * "empty"&mdash;devoid of archive data that could be
225   * {@linkplain #loadUnlessEmpty(Description, Connection) loaded}.
226   *
227   * @param description the {@link Description} describing the current
228   * JUnit test underway; ignored by this implementation but may be
229   * useful for subclasses; may be {@code null}
230   *
231   * @return {@code true} if this {@link H2Archive} is conceptually
232   * empty with regards to the supplied {@link Description}; {@code
233   * false} otherwise
234   *
235   * @see #loadUnlessEmpty(Description, Connection)
236   */
237  public boolean isEmpty(final Description description) {
238    try {
239      this.dataLock.readLock().lock();
240      return this.data == null || this.data.isEmpty();
241    } finally {
242      this.dataLock.readLock().unlock();
243    }
244  }
